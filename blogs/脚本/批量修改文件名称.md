## 批量修改文件名称

```js
let fs = require('fs');

function renameFile(path) {
    let files = fs.readdirSync(path);

    let num = 1001;
    files.forEach(async (item, index) => {
        let newName = 'work_' + num + '.jpg';

        if (item != 'Thumbs.db') {
            num++;
            await fs.renameSync(path + item, path + newName);
        }
    })
}

```
